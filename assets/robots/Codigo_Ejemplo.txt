#include <Wire.h>
#include <Adafruit_VL53L0X.h>

// ====================================================================
// 1. CONFIGURACION DE PINES
// ====================================================================
const int IN1 = 5;
const int IN2 = 6;
const int IN3 = 7;
const int IN4 = 8;

Adafruit_VL53L0X tof_Frente = Adafruit_VL53L0X(); 
Adafruit_VL53L0X tof_Der    = Adafruit_VL53L0X(); 
Adafruit_VL53L0X tof_Izq    = Adafruit_VL53L0X(); 

// ====================================================================
// 2. VARIABLES DE AJUSTE
// ====================================================================
int velBase = 130;      
int rangoMax = 100;     
int distFreno = 200;    
float gananciaBase = 30.0; // Ganancia cuando va a velocidad maxima

void setup() {
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);

  Serial.begin(115200);
  Wire.begin();

  if(!tof_Frente.begin(0x29)) Serial.println("Error ToF Frente");
  if(!tof_Der.begin(0x2B))    Serial.println("Error ToF Derecho");
  if(!tof_Izq.begin(0x2D))    Serial.println("Error ToF Izquierdo");

  delay(500);
}

void loop() {
  // Simulacion del delay de 3 sensores HC-SR04 (segun tu consulta)
  delay(60); 

  // --- 1. LECTURA DE SENSORES ---
  int d_frt = tof_Frente.readRange();
  int d_der = tof_Der.readRange();
  int d_izq = tof_Izq.readRange();

  // --- 2. LOGICA DE FRENADO FRONTAL ---
  float factorVel = 1.0;
  if (d_frt < distFreno) {
    factorVel = (float)d_frt / distFreno; 
  }
  // Mantenemos un minimo de 0.3 para que no se bloquee la division posterior
  factorVel = constrain(factorVel, 0.3, 1.0); 
  
  int velActual = velBase * factorVel;

  // --- 3. LOGICA DE SEGUIMIENTO (ERROR LATERAL) ---
  int errorRaw = d_izq - d_der; 
  float correccionNorm = (float)errorRaw / rangoMax;
  correccionNorm = constrain(correccionNorm, -1.0, 1.0);

  // --- 4. GANANCIA ADAPTATIVA ---
  // A medida que factorVel baja (hay algo enfrente), la ganancia sube.
  // Ejemplo: Si factorVel es 0.5, la ganancia se duplica.
  float gananciaDinamica = gananciaBase / factorVel;

  int vI = velActual - (correccionNorm * gananciaDinamica); 
  int vD = velActual + (correccionNorm * gananciaDinamica); 

  // --- 5. APLICAR MOTORES Y LOG ---
  aplicarMotores(vI, vD);

  Serial.print("F:"); Serial.print(d_frt);
  Serial.print(" | Fact:"); Serial.print(factorVel);
  Serial.print(" | Gan:"); Serial.print(gananciaDinamica);
  Serial.print(" | vI:"); Serial.print(vI);
  Serial.print(" | vD:"); Serial.println(vD);

  delay(30); 
}

// Funcion de control de motores por PWM en pines IN
void aplicarMotores(int vL, int vR) {
  // Motor Izquierdo
  vL = constrain(vL, -255, 255);
  if (vL >= 0) {
    analogWrite(IN1, vL);
    digitalWrite(IN2, LOW);
  } else {
    digitalWrite(IN1, LOW);
    analogWrite(IN2, abs(vL));
  }

  // Motor Derecho
  vR = constrain(vR, -255, 255);
  if (vR >= 0) {
    analogWrite(IN3, vR);
    digitalWrite(IN4, LOW);
  } else {
    digitalWrite(IN3, LOW);
    analogWrite(IN4, abs(vR));
  }
}
